import { BrowserRouter as Router, Routes, Route, Navigate, useLocation } from "react-router-dom";

// Centralized mock data to mimic a real backend database connection.

const SYSTEM_CONFIG = {
  version: "5.0.1",
  environment: "production",
  defaultTheme: "cyan",
  currency: "USD",
  taxRate: 0.08,
  shippingRate: 15.00,
  freeShippingThreshold: 500,
};


// ============================================================================
// GLOBAL AUTHENTICATION & STATE CONTEXT
// This is the absolute brain of the application. It handles local storage
// syncing, user sessions, theme preferences, and e-commerce databanks.
// ============================================================================
export const AuthContext = createContext();

export const GlobalProvider = ({ children }) => {
  // --- CORE STATE ---
  const [user, setUser] = useState(null);
  const [isInitializing, setIsInitializing] = useState(true);
  
  // --- E-COMMERCE STATE ---
  const [cart, setCart] = useState([]);
  const [wishlist, setWishlist] = useState([]);
  const [orderHistory, setOrderHistory] = useState([]);
  
  // --- UI/UX STATE ---
  const [theme, setTheme] = useState(SYSTEM_CONFIG.defaultTheme);
  const [systemLogs, setSystemLogs] = useState([]);

  // ------------------------------------------------------------------------
  // INITIALIZATION: Boot up the Neural Engine & Load Saved Data
  // ------------------------------------------------------------------------
  useEffect(() => {
    const bootSequence = async () => {
      try {
        console.log(`[ShopAI v${SYSTEM_CONFIG.version}] Initiating boot sequence...`);
        
        // 1. Check for Active User Session
        const savedSession = localStorage.getItem("shopai_user_session");
        let activeUser = null;
        
        if (savedSession) {
          activeUser = JSON.parse(savedSession);
          setUser(activeUser);
          console.log("[System] Valid user session restored.");
        } else {
          console.log("[System] No active session found. Running as Guest.");
        }

        // 2. Load Global / User-Specific Preferences
        // If a user is logged in, load THEIR specific cart/theme. If not, load guest data.
        const storagePrefix = activeUser ? `shopai_${activeUser.username}_` : "shopai_guest_";

        const savedTheme = localStorage.getItem(`${storagePrefix}theme`);
        if (savedTheme) setTheme(savedTheme);

        const savedCart = localStorage.getItem(`${storagePrefix}cart`);
        if (savedCart) setCart(JSON.parse(savedCart));

        const savedWishlist = localStorage.getItem(`${storagePrefix}wishlist`);
        if (savedWishlist) setWishlist(JSON.parse(savedWishlist));

        const savedOrders = localStorage.getItem(`${storagePrefix}orders`);
        if (savedOrders) setOrderHistory(JSON.parse(savedOrders));

      } catch (error) {
        console.error("[System Error] Boot sequence failed to parse storage:", error);
      } finally {
        // Add a slight delay to show the awesome loading screen
        setTimeout(() => {
          setIsInitializing(false);
        }, 1200);
      }
    };

    bootSequence();
  }, []);

  // ------------------------------------------------------------------------
  // DATA SYNCHRONIZATION: Save to LocalStorage whenever state changes
  // ------------------------------------------------------------------------
  useEffect(() => {
    if (isInitializing) return; // Prevent overwriting data before load completes

    const storagePrefix = user ? `shopai_${user.username}_` : "shopai_guest_";

    try {
      localStorage.setItem(`${storagePrefix}theme`, theme);
      localStorage.setItem(`${storagePrefix}cart`, JSON.stringify(cart));
      localStorage.setItem(`${storagePrefix}wishlist`, JSON.stringify(wishlist));
      localStorage.setItem(`${storagePrefix}orders`, JSON.stringify(orderHistory));
      
      // Update DOM for dynamic theme colors outside of React components if needed
      document.documentElement.setAttribute('data-theme', theme);
    } catch (error) {
      console.error("[Storage Error] Failed to sync data to local databanks:", error);
    }
  }, [theme, cart, wishlist, orderHistory, user, isInitializing]);

  // ------------------------------------------------------------------------
  // AUTHENTICATION METHODS
  // ------------------------------------------------------------------------
  const login = useCallback((userData) => {
    // Merge new user data with default permissions
    const completeUserData = {
      ...userData,
      id: `USR-${Math.floor(Math.random() * 90000) + 10000}`,
      loginTime: new Date().toISOString(),
      clearance: userData.clearance || "Level 1",
    };

    // Save to State and Storage
    setUser(completeUserData);
    localStorage.setItem("shopai_user_session", JSON.stringify(completeUserData));
    
    logSystemEvent(`User ${completeUserData.username} authenticated successfully.`);
    
    // Transfer Guest Cart to User Cart if applicable
    const guestCart = localStorage.getItem("shopai_guest_cart");
    if (guestCart && JSON.parse(guestCart).length > 0) {
      setCart(JSON.parse(guestCart));
      localStorage.removeItem("shopai_guest_cart"); // Clear guest cart
    }
  }, []);

  const logout = useCallback(() => {
    const username = user?.username || "Unknown";
    setUser(null);
    setCart([]);
    setWishlist([]);
    setTheme(SYSTEM_CONFIG.defaultTheme);
    localStorage.removeItem("shopai_user_session");
    logSystemEvent(`User ${username} terminated link.`);
  }, [user]);

  // ------------------------------------------------------------------------
  // E-COMMERCE METHODS (Advanced Cart Management)
  // ------------------------------------------------------------------------
  const processCheckout = useCallback((paymentDetails) => {
    if (!user) throw new Error("Authentication required for checkout.");
    if (cart.length === 0) throw new Error("Cannot process an empty databank.");

    const orderTotal = cart.reduce((sum, item) => sum + item.price * item.qty, 0);
    
    const newOrder = {
      orderId: `ORD-${Date.now()}-${Math.floor(Math.random() * 1000)}`,
      date: new Date().toISOString(),
      items: [...cart],
      total: orderTotal,
      status: "Processing Orbital Drop",
      paymentMethod: paymentDetails?.method || "Encrypted Credit",
    };

    // Save Order, Clear Cart
    setOrderHistory(prev => [newOrder, ...prev]);
    setCart([]);
    logSystemEvent(`Order ${newOrder.orderId} processed for $${orderTotal.toFixed(2)}`);
    
    return newOrder;
  }, [cart, user]);

  const clearCart = useCallback(() => {
    setCart([]);
    logSystemEvent("User purged active databank (cart cleared).");
  }, []);

  // ------------------------------------------------------------------------
  // SYSTEM UTILITIES
  // ------------------------------------------------------------------------
  const logSystemEvent = (message) => {
    const newLog = {
      id: Date.now(),
      timestamp: new Date().toLocaleTimeString(),
      message,
    };
    setSystemLogs(prev => [newLog, ...prev].slice(0, 50)); // Keep last 50 logs
    console.log(`[System Log] ${message}`);
  };

  // ------------------------------------------------------------------------
  // CONTEXT PAYLOAD
  // Memoized to prevent unnecessary re-renders of child components
  // ------------------------------------------------------------------------
  const contextPayload = useMemo(() => ({
    // Auth State
    user,
    login,
    logout,
    // E-Commerce State
    cart,
    setCart,
    wishlist,
    setWishlist,
    orderHistory,
    processCheckout,
    clearCart,
    // System Config
    systemConfig: SYSTEM_CONFIG,
    // UI State
    theme,
    setTheme,
    systemLogs,
  }), [
    user, login, logout, 
    cart, wishlist, orderHistory, processCheckout, clearCart, 
    theme, systemLogs
  ]);

  // ------------------------------------------------------------------------
  // RENDER LOADING SCREEN OR APP
  // ------------------------------------------------------------------------
  if (isInitializing) {
    return <SystemBootLoader />;
  }

  return (
    <AuthContext.Provider value={contextPayload}>
      {children}
    </AuthContext.Provider>
  );
};

  const activeColor = themes[theme].primary;

  // --- VALIDATION ENGINE ---
  const validate = (name, value, currentForm) => {
    let error = "";
    if (name === "identifier" && !value) error = "Identifier is required";
    if (name === "password" && !value) error = "Password is required";
    if (name === "magicEmail") {
      if (!value) error = "Email is required";
      else if (!/^\S+@\S+\.\S+$/.test(value)) error = "Invalid email format";
    }
    if (name === "newPassword") {
      if (!value) error = "Required";
      else if (value.length < 8) error = "Must be at least 8 characters";
    }
    if (name === "confirmPassword") {
      if (value !== currentForm.newPassword) error = "Passwords do not match";
    }
    return error;
  };

  const handleChange = (e) => {
    const { name, value } = e.target;
    setForm((prev) => {
      const newForm = { ...prev, [name]: value };
      setErrors((prevErr) => ({ ...prevErr, [name]: validate(name, value, newForm) }));
      return newForm;
    });
  };

  const triggerShake = (elementId = "auth-card") => {
    const card = document.getElementById(elementId);
    if (card) {
      card.classList.remove("shake-animation");
      void card.offsetWidth; // trigger reflow
      card.classList.add("shake-animation");
    }
  };

  const validateFields = (fields) => {
    let isValid = true;
    const newErrors = {};
    fields.forEach((field) => {
      const error = validate(field, form[field], form);
      if (error) {
        newErrors[field] = error;
        isValid = false;
      }
    });
    setErrors((prev) => ({ ...prev, ...newErrors }));
    if (!isValid) triggerShake();
    return isValid;
  };

  // --- PASSWORD STRENGTH ENGINE ---
  const calculateStrength = (pass) => {
    let score = 0;
    if (!pass) return { score: 0, label: "", color: "transparent", width: "0%" };
    if (pass.length > 7) score += 1;
    if (/[A-Z]/.test(pass)) score += 1;
    if (/[0-9]/.test(pass)) score += 1;
    if (/[^A-Za-z0-9]/.test(pass)) score += 1;
    if (score <= 1) return { label: "Weak", color: "#f43f5e", width: "25%" };
    if (score === 2) return { label: "Fair", color: "#fbbf24", width: "50%" };
    if (score === 3) return { label: "Good", color: "#34d399", width: "75%" };
    return { label: "Strong", color: activeColor, width: "100%" };
  };

  // --- OTP INPUT LOGIC ---
  const handleOtpChange = (index, e) => {
    const value = e.target.value;
    if (isNaN(value)) return;
    const newOtp = [...otpValues];
    newOtp[index] = value.substring(value.length - 1); // Keep only last digit
    setOtpValues(newOtp);
    // Auto-advance
    if (value && index < 5) otpRefs.current[index + 1].focus();
  };

  const handleOtpKeyDown = (index, e) => {
    if (e.key === "Backspace" && !otpValues[index] && index > 0) {
      otpRefs.current[index - 1].focus();
    }
  };

  const handleOtpPaste = (e) => {
    e.preventDefault();
    const pastedData = e.clipboardData.getData("text").slice(0, 6).split("");
    if (pastedData.some(isNaN)) {
      addToast("Invalid code format. Numbers only.", "error");
      return;
    }
    const newOtp = [...otpValues];
    pastedData.forEach((char, i) => { if (i < 6) newOtp[i] = char; });
    setOtpValues(newOtp);
    const focusIndex = Math.min(pastedData.length, 5);
    otpRefs.current[focusIndex].focus();
  };

  // --- GLOBAL AUTHENTICATION INJECTOR ---
  const executeGlobalLogin = (identifier, backendUser = null) => {
    const normalizedEmail = identifier.includes('@') ? identifier : `${identifier}@shopai.net`;
    const userData = backendUser ? {
      ...backendUser,
      clearance: "Level 4",
      username: backendUser.username || normalizedEmail.split('@')[0],
      email: backendUser.email || normalizedEmail,
    } : {
      username: normalizedEmail.split('@')[0] || "Agent",
      email: normalizedEmail,
      clearance: "Level 4"
    };
    login(userData); // Set global context
    localStorage.setItem("shopai_auth_token", backendUser?.token || "");
    setAuthStep("dashboard");
    
    // Auto-navigate to store after brief success message
    setTimeout(() => {
      navigate('/shop');
    }, 1500);
  };

  // --- SUBMISSION HANDLERS ---
  const mockApiCall = (time = 1500) => new Promise((res) => setTimeout(res, time));

  const handleLogin = async (e) => {
    e.preventDefault();
    if (!validateFields(["identifier", "password"])) return;
    setIsSubmitting(true);
    await mockApiCall(1200);
    setIsSubmitting(false);
    
    // Go to 2FA screen
    addToast("Credentials verified. Additional security required.", "info");
    setAuthStep("2fa");
    setOtpValues(["", "", "", "", "", ""]);
  };

  const handleMagicLink = async (e) => {
    e.preventDefault();
    if (!validateFields(["magicEmail"])) return;
    setIsSubmitting(true);
    await mockApiCall(1500);
    setIsSubmitting(false);
    addToast(`Magic link sent to ${form.magicEmail}`, "success");
    setForm({ ...form, magicEmail: "" });
  };

  const handlePasskey = async () => {
    setPasskeyStatus("scanning");
    await mockApiCall(2000);
    // 80% chance of success for realism
    if (Math.random() > 0.2) {
      setPasskeyStatus("success");
      addToast("Biometrics verified.", "success");
      executeGlobalLogin("BioUser");
    } else {
      setPasskeyStatus("error");
      addToast("Biometric mismatch. Try again.", "error");
      setTimeout(() => setPasskeyStatus("idle"), 2000);
    }
  };

  const handleVerify2FA = async (e) => {
    e.preventDefault();
    const code = otpValues.join("");
    if (code.length < 6) {
      triggerShake();
      addToast("Please enter the complete 6-digit code.", "error");
      return;
    }
    setIsSubmitting(true);
    await mockApiCall(500);

    if (code === "000000") {
      addToast("Invalid security code.", "error");
      setOtpValues(["", "", "", "", "", ""]);
      otpRefs.current[0].focus();
      triggerShake();
      setIsSubmitting(false);
    } else {
      try {
        const email = form.identifier.includes("@") ? form.identifier : `${form.identifier}@shopai.net`;
        const response = await fetch(`${API_BASE_URL}/api/login`, {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ email, password: form.password }),
        });
        const data = await response.json();

        if (!response.ok) {
          throw new Error(data.error || "Login failed");
        }

        addToast("Authentication complete. Uplinking...", "success");
        executeGlobalLogin(form.identifier, { ...data.user, token: data.token });
      } catch (error) {
        addToast(error.message || "Unable to reach auth server.", "error");
        triggerShake();
      } finally {
        setIsSubmitting(false);
      }
    }
  };

  const handleSocialLogin = (provider) => {
    addToast(`Authenticating with ${provider} secure servers...`, "info");
    setIsSubmitting(true);
    setTimeout(() => {
      setIsSubmitting(false);
      executeGlobalLogin(`${provider}_User`);
    }, 1500);
  };

  // --- FORGOT PASSWORD FLOW ---
  const handleForgotEmailSubmit = async (e) => {
    e.preventDefault();
    if (!validateFields(["magicEmail"])) return;
    setIsSubmitting(true);
    await mockApiCall(1000);
    setIsSubmitting(false);
    addToast("Recovery code sent to email.", "success");
    setAuthStep("forgot-otp");
    setOtpValues(["", "", "", "", "", ""]);
  };

  const handleForgotOtpSubmit = async (e) => {
    e.preventDefault();
    if (otpValues.join("").length < 6) { triggerShake(); return; }
    setIsSubmitting(true);
    await mockApiCall(800);
    setIsSubmitting(false);
    setAuthStep("forgot-reset");
  };

  const handleResetPassword = async (e) => {
    e.preventDefault();
    if (!validateFields(["newPassword", "confirmPassword"])) return;
    setIsSubmitting(true);
    await mockApiCall(1500);
    setIsSubmitting(false);
    addToast("Password successfully encrypted and updated.", "success");
    setAuthStep("login");
    setForm({ ...form, password: "", newPassword: "", confirmPassword: "" });
  };

  // --- INLINE SVGS ---
  const Icons = {
    user: <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><path d="M20 21v-2a4 4 0 0 0-4-4H8a4 4 0 0 0-4 4v2"/><circle cx="12" cy="7" r="4"/></svg>,
    mail: <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><rect x="2" y="4" width="20" height="16" rx="2"/><path d="m22 7-8.97 5.7a1.94 1.94 0 0 1-2.06 0L2 7"/></svg>,
    lock: <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><rect x="3" y="11" width="18" height="11" rx="2" ry="2"/><path d="M7 11V7a5 5 0 0 1 10 0v4"/></svg>,
    eye: <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><path d="M2 12s3-7 10-7 10 7 10 7-3 7-10 7-10-7-10-7Z"/><circle cx="12" cy="12" r="3"/></svg>,
    eyeOff: <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><path d="M9.88 9.88a3 3 0 1 0 4.24 4.24"/><path d="M10.73 5.08A10.43 10.43 0 0 1 12 5c7 0 10 7 10 7a13.16 13.16 0 0 1-1.67 2.68"/><path d="M6.61 6.61A13.526 13.526 0 0 0 2 12s3 7 10 7a9.74 9.74 0 0 0 5.39-1.61"/><line x1="2" y1="2" x2="22" y2="22"/></svg>,
    fingerprint: <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><path d="M2 12C2 6.48 6.48 2 12 2s10 4.48 10 10"/><path d="M5 15c0-3.87 3.13-7 7-7s7 3.13 7 7"/><path d="M8 18c0-2.21 1.79-4 4-4s4 1.79 4 4"/><path d="M12 21v-1"/></svg>,
    shield: <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><path d="M12 22s8-4 8-10V5l-8-3-8 3v7c0 6 8 10 8 10z"/></svg>,
    sparkles: <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><path d="m12 3-1.9 5.8a2 2 0 0 1-1.3 1.3L3 12l5.8 1.9a2 2 0 0 1 1.3 1.3L12 21l1.9-5.8a2 2 0 0 1 1.3-1.3L21 12l-5.8-1.9a2 2 0 0 1-1.3-1.3L12 3Z"/></svg>,
    smartphone: <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><rect x="5" y="2" width="14" height="20" rx="2" ry="2"/><line x1="12" y1="18" x2="12.01" y2="18"/></svg>,
    arrowLeft: <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><line x1="19" y1="12" x2="5" y2="12"/><polyline points="12 19 5 12 12 5"/></svg>,
    check: <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><polyline points="20 6 9 17 4 12"/></svg>,
    alert: <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><circle cx="12" cy="12" r="10"/><line x1="12" y1="8" x2="12" y2="12"/><line x1="12" y1="16" x2="12.01" y2="16"/></svg>,
    x: <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><line x1="18" y1="6" x2="6" y2="18"/><line x1="6" y1="6" x2="18" y2="18"/></svg>,
    google: <svg viewBox="0 0 24 24" fill="currentColor"><path d="M22.56 12.25c0-.78-.07-1.53-.2-2.25H12v4.26h5.92c-.26 1.37-1.04 2.53-2.21 3.31v2.77h3.57c2.08-1.92 3.28-4.74 3.28-8.09z" fill="#4285F4"/><path d="M12 23c2.97 0 5.46-.98 7.28-2.66l-3.57-2.77c-.98.66-2.23 1.06-3.71 1.06-2.86 0-5.29-1.93-6.16-4.53H2.18v2.84C3.99 20.53 7.7 23 12 23z" fill="#34A853"/><path d="M5.84 14.09c-.22-.66-.35-1.36-.35-2.09s.13-1.43.35-2.09V7.07H2.18C1.43 8.55 1 10.22 1 12s.43 3.45 1.18 4.93l2.85-2.22.81-.62z" fill="#FBBC05"/><path d="M12 5.38c1.62 0 3.06.56 4.21 1.64l3.15-3.15C17.45 2.09 14.97 1 12 1 7.7 1 3.99 3.47 2.18 7.07l3.66 2.84c.87-2.6 3.3-4.53 6.16-4.53z" fill="#EA4335"/></svg>,
    apple: <svg viewBox="0 0 24 24" fill="currentColor"><path d="M12.15 2c-.08 1.83 1.25 3.52 2.92 3.65.25-1.74-1.2-3.48-2.92-3.65zm4.82 4.2c-1.63-.05-3.15.93-3.95.93-.82 0-2.12-.9-3.48-.88-1.77.03-3.41 1.03-4.32 2.62-1.85 3.2-.47 7.93 1.33 10.53.88 1.28 1.93 2.72 3.33 2.67 1.35-.05 1.87-.87 3.5-.87 1.62 0 2.1.87 3.52.84 1.45-.03 2.37-1.32 3.23-2.6.98-1.44 1.38-2.85 1.4-2.92-.03-.02-2.73-1.05-2.76-4.18-.03-2.62 2.13-3.88 2.23-3.93-1.23-1.8-3.15-2.05-3.97-2.11z"/></svg>,
    github: <svg viewBox="0 0 24 24" fill="currentColor"><path d="M12 2A10 10 0 0 0 8.84 21.5c.5.08.66-.23.66-.5v-1.69c-2.77.6-3.36-1.34-3.36-1.34-.45-1.15-1.1-1.46-1.1-1.46-.91-.62.07-.6.07-.6 1 .07 1.53 1.03 1.53 1.03.88 1.52 2.32 1.08 2.88.83.09-.64.35-1.08.63-1.33-2.22-.25-4.55-1.11-4.55-4.92 0-1.09.39-1.98 1.03-2.68-.1-.25-.45-1.27.1-2.64 0 0 .84-.27 2.75 1.02a9.58 9.58 0 0 1 5 0c1.9-1.29 2.75-1.02 2.75-1.02.55 1.37.2 2.39.1 2.64.64.7 1.03 1.59 1.03 2.68 0 3.82-2.34 4.66-4.57 4.91.36.31.68.92.68 1.85v2.75c0 .28.16.59.67.5A10 10 0 0 0 12 2z"/></svg>,
  };

  // --- REUSABLE INPUT COMPONENT ---
  const InputField = ({ label, name, type = "text", icon, isPassword }) => {
    const hasError = errors[name];
    const actualType = isPassword ? (showPassword ? "text" : "password") : type;

    return (
      <div className="input-wrapper">
        <input
          type={actualType} name={name} id={name} value={form[name]} onChange={handleChange}
          className={`glass-input ${hasError ? "error" : ""} ${form[name] ? "has-value" : ""}`} placeholder=" "
        />
        <div className="icon-container">{Icons[icon]}</div>
        <label htmlFor={name} className="floating-label">{label}</label>

        {isPassword && (
          <button type="button" className="eye-btn" onClick={() => setShowPassword(!showPassword)} tabIndex="-1">
            {showPassword ? Icons.eyeOff : Icons.eye}
          </button>
        )}
        {hasError && <span className="error-text slide-up">{errors[name]}</span>}
      </div>
    );
  };

  // ==========================================
  // RENDER SECTIONS
  // ==========================================
  const renderSocialLogins = () => (
    <div className="social-container slide-up" style={{ animationDelay: '0.2s' }}>
      <div className="divider"><span>or continue with</span></div>
      <div className="social-buttons">
        <button type="button" className="social-btn" onClick={() => handleSocialLogin('Google')} disabled={isSubmitting}>{Icons.google}</button>
        <button type="button" className="social-btn" onClick={() => handleSocialLogin('Apple')} disabled={isSubmitting}>{Icons.apple}</button>
        <button type="button" className="social-btn" onClick={() => handleSocialLogin('GitHub')} disabled={isSubmitting}>{Icons.github}</button>
      </div>
    </div>
  );

  const renderTabs = () => (
    <div className="auth-tabs">
      <button className={`tab-btn ${activeTab === 'password' ? 'active' : ''}`} onClick={() => setActiveTab('password')}>Password</button>
      <button className={`tab-btn ${activeTab === 'magic' ? 'active' : ''}`} onClick={() => setActiveTab('magic')}>Magic Link</button>
      <button className={`tab-btn ${activeTab === 'passkey' ? 'active' : ''}`} onClick={() => setActiveTab('passkey')}>Passkey</button>
      <div className="tab-indicator" style={{ 
        transform: `translateX(${activeTab === 'password' ? '0' : activeTab === 'magic' ? '100%' : '200%'})` 
      }}></div>
    </div>
  );

  // VIEW: MAIN LOGIN
  const renderLoginView = () => (
    <div id="auth-card" className="glass-card fadeIn">
      <div className="header">
        <div className="logo-glow">{Icons.shield}</div>
        <h2>Access Portal</h2>
        <p>Identify yourself to continue</p>
      </div>

      {renderTabs()}

      {/* TAB: PASSWORD */}
      {activeTab === 'password' && (
        <form onSubmit={handleLogin} className="tab-content fadeIn" noValidate>
          <InputField name="identifier" label="Email or Username" icon="user" />
          <InputField name="password" label="Password" icon="lock" isPassword={true} />
          
          <div className="form-options">
            <label className="checkbox-wrap">
              <input type="checkbox" />
              <span className="checkmark"></span>
              Remember device
            </label>
            <span className="link-text" onClick={() => { setAuthStep('forgot-email'); setErrors({}); }}>Forgot password?</span>
          </div>

          <button type="submit" className="neon-btn" disabled={isSubmitting}>
            {isSubmitting ? <span className="spinner"></span> : "Authenticate"}
          </button>
          {renderSocialLogins()}
        </form>
      )}

      {/* TAB: MAGIC LINK */}
      {activeTab === 'magic' && (
        <form onSubmit={handleMagicLink} className="tab-content fadeIn" noValidate>
          <div className="info-box">
            {Icons.sparkles}
            <p>We'll email you a magic link for a password-free sign in.</p>
          </div>
          <InputField name="magicEmail" label="Email Address" icon="mail" type="email" />
          <button type="submit" className="neon-btn" disabled={isSubmitting}>
            {isSubmitting ? <span className="spinner"></span> : "Send Magic Link"}
          </button>
          {renderSocialLogins()}
        </form>
      )}

      {/* TAB: PASSKEY */}
      {activeTab === 'passkey' && (
        <div className="tab-content fadeIn text-center">
           <div className="info-box mb-6">
            {Icons.fingerprint}
            <p>Use your device's biometric sensor or security key for instant access.</p>
          </div>
          
          <div className={`biometric-scanner ${passkeyStatus}`}>
             <div className="scan-line"></div>
             {passkeyStatus === 'success' ? Icons.check : passkeyStatus === 'error' ? Icons.x : Icons.fingerprint}
          </div>

          <button type="button" className="neon-btn mt-6" onClick={handlePasskey} disabled={passkeyStatus === 'scanning'}>
            {passkeyStatus === 'scanning' ? "Awaiting Biometrics..." : "Scan Passkey"}
          </button>
        </div>
      )}

      {/* NAVIGATION TO SIGNUP */}
      <div className="footer-text">
        No clearance? <span className="link-text font-bold" onClick={() => navigate('/signup')}>Request Access</span>
      </div>
    </div>
  );

  // VIEW: 2FA MFA
  const render2FAView = () => (
    <form id="auth-card" className="glass-card fadeIn text-center" onSubmit={handleVerify2FA}>
      <button type="button" className="back-btn" onClick={() => setAuthStep('login')}>{Icons.arrowLeft}</button>
      
      <div className="header mt-4">
        <div className="logo-glow pulse">{Icons.smartphone}</div>
        <h2>2-Step Verification</h2>
        <p className="px-4">Enter the 6-digit security sequence generated by your authenticator app.</p>
      </div>

      <div className="otp-container mt-6">
        {otpValues.map((val, index) => (
          <input
            key={index} ref={(el) => (otpRefs.current[index] = el)} type="text" inputMode="numeric" maxLength={1} value={val}
            onChange={(e) => handleOtpChange(index, e)} onKeyDown={(e) => handleOtpKeyDown(index, e)}
            onPaste={index === 0 ? handleOtpPaste : null} className="otp-input" autoFocus={index === 0}
          />
        ))}
      </div>

      <button type="submit" className="neon-btn mt-8" disabled={isSubmitting}>
        {isSubmitting ? <span className="spinner"></span> : "Verify Identity"}
      </button>

      <div className="footer-text mt-6">
        Lost access to device? <span className="link-text">Use recovery code</span>
      </div>
    </form>
  );

  // VIEW: FORGOT PASSWORD FLOW
  const renderForgotFlow = () => {
    return (
      <div id="auth-card" className="glass-card fadeIn">
        <button type="button" className="back-btn" onClick={() => setAuthStep('login')}>{Icons.arrowLeft}</button>
        
        {authStep === 'forgot-email' && (
          <form onSubmit={handleForgotEmailSubmit} className="fadeIn mt-4" noValidate>
            <div className="header">
              <h2>Recover Access</h2>
              <p>Enter your linked email address</p>
            </div>
            <InputField name="magicEmail" label="Registered Email" icon="mail" type="email" />
            <button type="submit" className="neon-btn mt-4" disabled={isSubmitting}>
              {isSubmitting ? <span className="spinner"></span> : "Search Records"}
            </button>
          </form>
        )}

        {authStep === 'forgot-otp' && (
          <form onSubmit={handleForgotOtpSubmit} className="fadeIn text-center mt-4" noValidate>
            <div className="header">
              <h2>Verify Email</h2>
              <p>Enter the 6-digit code sent to <strong style={{color: activeColor}}>{form.magicEmail}</strong></p>
            </div>
            <div className="otp-container mt-6">
              {otpValues.map((val, index) => (
                <input
                  key={index} ref={(el) => (otpRefs.current[index] = el)} type="text" inputMode="numeric" maxLength={1} value={val}
                  onChange={(e) => handleOtpChange(index, e)} onKeyDown={(e) => handleOtpKeyDown(index, e)}
                  onPaste={index === 0 ? handleOtpPaste : null} className="otp-input" autoFocus={index === 0}
                />
              ))}
            </div>
            <button type="submit" className="neon-btn mt-8" disabled={isSubmitting}>
              {isSubmitting ? <span className="spinner"></span> : "Confirm Code"}
            </button>
            <div className="footer-text mt-6">Didn't receive it? <span className="link-text">Resend</span></div>
          </form>
        )}

        {authStep === 'forgot-reset' && (
          <form onSubmit={handleResetPassword} className="fadeIn mt-4" noValidate>
            <div className="header">
              <h2>New Credentials</h2>
              <p>Establish a secure passphrase</p>
            </div>
            
            <InputField name="newPassword" label="New Password" icon="lock" isPassword={true} />
            
            {form.newPassword && (
              <div className="strength-meter-wrap slide-up">
                <div className="strength-bars">
                  {[1, 2, 3, 4].map((level) => {
                    const strength = calculateStrength(form.newPassword);
                    const isActive = strength.score >= level || (level === 1 && form.newPassword.length > 0);
                    return (
                      <div key={level} className="strength-bar-chunk" style={{ backgroundColor: isActive ? strength.color : 'rgba(255,255,255,0.1)' }}></div>
                    );
                  })}
                </div>
                <div className="strength-text" style={{ color: calculateStrength(form.newPassword).color }}>
                  {calculateStrength(form.newPassword).label}
                </div>
              </div>
            )}

            <InputField name="confirmPassword" label="Confirm Password" icon="shield" isPassword={true} />

              {[1, 2, 3, 4].map(step => (
                <div key={step} className={`step-dot ${signupStep === step ? 'active' : ''} ${signupStep > step ? 'completed' : ''}`}>
                  {signupStep > step ? Icons.check : step}
                </div>
              ))}
            </div>
          </div>

          {/* STEP 1: CREDENTIALS */}
          {signupStep === 1 && (
            <div className="step-content fadeIn">
              <InputField name="email" label="Encrypted Email" icon="mail" type="email" />
              <InputField name="username" label="Codename" icon="user" />
              <InputField name="password" label="Passphrase" icon="lock" isPassword={true} />
              
              {/* Strength Meter */}
              {form.password && (
                <div className="strength-meter-wrap slide-up">
                  <div className="strength-bars">
                    {[1, 2, 3, 4].map((level) => {
                      const strength = calculateStrength(form.password);
                      const isActive = strength.score >= level || (level === 1 && form.password.length > 0);
                      return (
                        <div key={level} className="strength-bar-chunk" style={{ backgroundColor: isActive ? strength.color : 'rgba(255,255,255,0.1)' }}></div>
                      );
                    })}
                  </div>
                  <div className="strength-text" style={{ color: calculateStrength(form.password).color }}>
                    {calculateStrength(form.password).label}
                  </div>
                </div>
              )}

              <InputField name="confirm" label="Verify Passphrase" icon="shield" isPassword={true} />
              
              <button type="button" className="neon-btn mt-4" onClick={nextStep}>
                Proceed to Biometrics {Icons.arrowRight}
              </button>
            </div>
          )}

          {/* STEP 2: PROFILE & AVATAR */}
          {signupStep === 2 && (
            <div className="step-content fadeIn">
              <span className="avatar-label">Select Holographic Avatar</span>
              <div className="avatar-grid">
                {avatars.map(av => (
                  <div key={av} className={`avatar-option ${form.avatar === av ? 'active' : ''}`} onClick={() => setForm({...form, avatar: av})}>
                    <img src={`https://api.dicebear.com/7.x/bottts/svg?seed=${av}&backgroundColor=transparent`} alt={av} />
                  </div>
                ))}
              </div>

              <div className="row-inputs">
                <InputField name="firstName" label="First Name" icon="user" />
                <InputField name="lastName" label="Last Name" icon="user" />
              </div>
              
              <InputField name="dob" label="Date of Activation" icon="calendar" type="date" />
              <InputField name="phone" label="Comm-Link (Optional)" icon="phone" type="tel" />

              <div className="checkbox-wrapper mt-4">
                <input type="checkbox" id="termsAccepted" name="termsAccepted" checked={form.termsAccepted} onChange={handleChange} />
                <label htmlFor="termsAccepted">
                  I accept the <span className="link-text" onClick={(e) => { e.preventDefault(); setShowTerms(true); }}>Terms of Service</span> and Neural Guidelines.
                </label>
              </div>
              {touched.termsAccepted && errors.termsAccepted && <span className="error-text block">{errors.termsAccepted}</span>}

              <div className="btn-group">
                <button type="button" className="ghost-btn" onClick={prevStep}>{Icons.arrowLeft}</button>
                <button type="button" className="neon-btn flex-1" onClick={nextStep}>Configure Tech {Icons.arrowRight}</button>
              </div>
            </div>
          )}

          {/* STEP 3: TECH PREFERENCES */}
          {signupStep === 3 && (
            <div className="step-content fadeIn text-center">
              <p style={{color: 'var(--text-muted)', marginBottom: '24px'}}>Select your hardware interests to calibrate our AI engine.</p>
              
              <div className="tech-grid">
                {techOptions.map(tech => (
                  <div key={tech.id} className={`tech-card ${form.preferredTech.includes(tech.id) ? 'active' : ''}`} onClick={() => toggleTech(tech.id)}>
                    {Icons[tech.icon]}
                    <span>{tech.label}</span>
                  </div>
                ))}
              </div>

              <div className="checkbox-wrapper" style={{ justifyContent: 'center' }}>
                <input type="checkbox" id="newsletter" name="newsletter" checked={form.newsletter} onChange={handleChange} />
                <label htmlFor="newsletter">Subscribe to orbital supply drops.</label>
              </div>

              <div className="btn-group">
                <button type="button" className="ghost-btn" onClick={prevStep}>{Icons.arrowLeft}</button>
                <button type="button" className="neon-btn flex-1" onClick={nextStep}>Finalize Uplink {Icons.arrowRight}</button>
              </div>
            </div>
          )}

          {/* STEP 4: 2FA VERIFICATION */}
          {signupStep === 4 && (
            <div className="step-content fadeIn text-center">
              <div className="otp-icon-wrap">{Icons.shield}</div>
              <h3>Security Verification</h3>
              <p style={{color: 'var(--text-muted)', margin: '12px 0 32px', lineHeight: '1.6'}}>
                Enter the 6-digit encryption sequence sent to <br/><strong style={{color: 'white'}}>{form.email}</strong>
              </p>
              
              <div className="otp-container">
                {otpValues.map((val, index) => (
                  <input
                    key={index} ref={(el) => (otpRefs.current[index] = el)} type="text" inputMode="numeric" maxLength={1} value={val}
                    onChange={(e) => handleOtpChange(index, e)} onKeyDown={(e) => handleOtpKeyDown(index, e)}
                    onPaste={index === 0 ? handleOtpPaste : null} className="otp-input" autoFocus={index === 0}
                  />
                ))}
              </div>

              <div className="btn-group mt-4">
                <button type="button" className="ghost-btn" onClick={prevStep} disabled={isSubmitting}>{Icons.arrowLeft}</button>
                <button type="submit" className="neon-btn flex-1" disabled={isSubmitting}>
                  {isSubmitting ? <span className="spinner"></span> : "Establish Link"}
                </button>
              </div>
              
              <p style={{marginTop: '24px', fontSize: '13px', color: 'var(--text-muted)'}}>
                Signal lost? <span className="link-text">Resend Code</span>
              </p>
            </div>
          )}

          <div className="switch-view-text">
            Already registered? <span className="link-text" onClick={() => navigate('/login')}>Authenticate</span>
          </div>
        </form>
      </div>

      {renderTermsModal()}

      {/* Global Toast Layer */}
      <div className="toast-container">
        {toasts.map((toast) => (
          <div key={toast.id} className={`toast ${toast.type}`}>
            {toast.type === "success" ? Icons.check : toast.type === "error" ? Icons.x : Icons.alert}
            <span>{toast.message}</span>
          </div>
        ))}
      </div>
    </>
  );
};

export default SignupPage;

const express = require('express');
const mongoose = require('mongoose');
const cors = require('cors');
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');
require('dotenv').config();

const app = express();
app.use(express.json());
app.use(cors());

// Database connection
const MONGO_URI = process.env.MONGO_URI || 'mongodb://127.0.0.1:27017/myfullstackapp';
const JWT_SECRET = process.env.JWT_SECRET || 'secretkey123';
const memoryUsers = [];
let dbConnected = false;

mongoose.connect(MONGO_URI).then(() => console.log('Connected to MongoDB'))
  .catch(err => console.error('Failed to connect to MongoDB', err));

mongoose.connection.on('connected', () => {
  dbConnected = true;
});

mongoose.connection.on('disconnected', () => {
  dbConnected = false;
});

// User Model
const UserSchema = new mongoose.Schema({
  username: { type: String, required: true, unique: true },
  email: { type: String, required: true, unique: true },
  password: { type: String, required: true },
  firstName: { type: String, default: '' },
  lastName: { type: String, default: '' }
});
const User = mongoose.model('User', UserSchema);

app.get('/api/health', (_req, res) => {
  res.json({ status: 'ok', service: 'auth-api' });
});

// Signup Route
app.post('/api/signup', async (req, res) => {
  try {
    const { email, username, password, firstName, lastName } = req.body;
    if (!email || !password || !username) return res.status(400).json({ error: 'Email, username and password are required' });

    if (dbConnected) {
      const existingEmail = await User.findOne({ email });
      if (existingEmail) return res.status(400).json({ error: 'Email already exists' });

      const existingUsername = await User.findOne({ username });
      if (existingUsername) return res.status(400).json({ error: 'Username already exists' });
    } else {
      const existingEmail = memoryUsers.find((u) => u.email === email);
      if (existingEmail) return res.status(400).json({ error: 'Email already exists' });

      const existingUsername = memoryUsers.find((u) => u.username === username);
      if (existingUsername) return res.status(400).json({ error: 'Username already exists' });
    }

    const hashedPassword = await bcrypt.hash(password, 10);
    let user;

    if (dbConnected) {
      user = new User({ email, username, password: hashedPassword, firstName, lastName });
      await user.save();
    } else {
      user = {
        _id: `MEM-${Date.now()}-${Math.floor(Math.random() * 1000)}`,
        email,
        username,
        password: hashedPassword,
        firstName: firstName || '',
        lastName: lastName || '',
      };
      memoryUsers.push(user);
    }

    const token = jwt.sign({ id: user._id }, JWT_SECRET, { expiresIn: '1h' });
    
    res.status(201).json({
      message: 'User created successfully',
      token,
      user: {
        id: user._id,
        username: user.username,
        email: user.email,
        firstName: user.firstName,
        lastName: user.lastName,
      }
    });
  } catch (err) {
    res.status(500).json({ error: 'Server error during signup' });
  }
});

// Login Route
app.post('/api/login', async (req, res) => {
  try {
    const { email, password } = req.body;
    const user = dbConnected
      ? await User.findOne({ email })
      : memoryUsers.find((u) => u.email === email);

    if (!user) return res.status(400).json({ error: 'Invalid credentials' });
    
    const isMatch = await bcrypt.compare(password, user.password);
    if (!isMatch) return res.status(400).json({ error: 'Invalid credentials' });
    
    const token = jwt.sign({ id: user._id }, JWT_SECRET, { expiresIn: '1h' });
    res.json({
      token,
      message: 'Logged in successfully',
      user: {
        id: user._id,
        username: user.username,
        email: user.email,
        firstName: user.firstName,
        lastName: user.lastName,
      }
    });
  } catch (err) {
    res.status(500).json({ error: 'Server error during login' });
  }
});

const PORT = process.env.PORT || 5000;
app.listen(PORT, () => console.log(`Server running on port ${PORT}`));
